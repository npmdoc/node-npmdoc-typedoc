    findReflectionByName = function (arg) {
    var names = Array.isArray(arg) ? arg : arg.split('.');
    var name = names.pop();
    search: for (var key in this.reflections) {
        var reflection = this.reflections[key];
        if (reflection.name !== name) {

        var target = reflection;
